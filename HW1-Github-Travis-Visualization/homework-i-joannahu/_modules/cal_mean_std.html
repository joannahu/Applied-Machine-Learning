import numpy as np


[docs]def meanstd(data, axis):
    """Compute the mean and the standard deviation of an array.

    Parameters
    ----------
    data : array_like
           input matrix
    axis:  int
           0 for flattened array; 1 for vertical array; NULL for all entries

    Returns
    -------
    m: array_like or float
       mean of the input array
    s: array_like or float
       standard deviation of the input array
    """
    m = np.mean(data, axis)
    s = np.std(data,axis)
    return m,s
